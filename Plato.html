    # File lib/alimentos/plato.rb, line 26
def add_food(gramos,food)
        @gramos+=gramos
        gramos = gramos/100
        food.*(gramos) #Alimento.*(value)
        lista.insert_head(food)
        @gramos_p += food.prot
        @gramos_l += food.lip
        @gramos_c += food.carbo
        @gramos_t += @gramos_p + @gramos_l + @gramos_c
        @porcent_p = (@gramos_p/@gramos_t)*100
        @porcent_l = (@gramos_l/@gramos_t)*100
        @porcent_c = (@gramos_c/@gramos_t)*100
        @kcal += @gramos_p*4 + @gramos_l*9 + @gramos_c*4
end
